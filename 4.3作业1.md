# 汉诺塔
```
#include <stdio.h>
void hanio(int n,char A,char B,char C);

int main(){
    int i=2;
    char A='1',B='2',C='3';
    hanio(i, A, B, C);
}

void hanio(int n,char A,char B,char C){
    
    if(n==1){
        printf("%c-->%c\n",A,B);
        
    }
    
    else{
        hanio(n-1, A, C, B);
        printf("%c-->%c\n",A,B);
        
        hanio(n-1, C, B, A);
        
    }
    return;
    ```
